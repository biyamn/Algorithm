```python
# 재귀 함수의 단점: n이 증가하면 시간 복잡도(O(2n))가 가파르게 증가
def solution(n):
    if n == 0:
        return 0
    elif n == 1 or n == 2:
        return 1
    else:
        return (solution(n - 1) + solution(n - 2))%1234567
```

```python
# for은 O(N)

def solution(n):
    # 아래 두 줄은 생략 가능했다!
    # if n == 2:
    #     return 1
    
		# F(0) = 0, F(1) = 1 
    a, b = 0, 1
    
    # n이 5라면 range(n-1)로 4번 돈다.
    for x in range(n - 1):
      # x가 0일 때 b는 0+1=1
      # x가 1일 때 b는 1+1=2
      # x가 2일 때 b는 1+2=3
      # x가 3일 때 b는 2+3=5
        a, b = b, a + b
        
    return b % 1234567
```
